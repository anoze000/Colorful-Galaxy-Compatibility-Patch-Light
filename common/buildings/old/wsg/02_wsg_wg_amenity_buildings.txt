building_wsg_luxury_residence = {
	base_buildtime = 240
	category = amenity
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_authority = auth_warshipgirls	
			NOT = { has_origin = origin_wsg_ice_breaking }
		}
		NOT = { has_modifier = slave_colony }
	}
	building_sets = {
		government
		urban
		entertainment
		resort
	}
	allow = {
	}
		
	planet_modifier = {
		planet_housing_add = 1000
		planet_amenities_add = 1500
		job_wsg_mahjong_add = 400
	}
	triggered_desc = {
		text = job_wsg_mahjong_effect_desc
	}	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_domination_imperious_architecture
			}					
		}
		modifier = {
			planet_housing_add = 100
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			energy = 5
		}
	}	
	
	ai_weight = {
		factor = 10
		modifier = {
			factor = 10
			free_amenities <= 0
		}
	}
	
	ai_resource_production = {
		energy = 5
		trigger = { always = yes }
	}
}

building_wsg_bathhouse = {
	base_buildtime = 360
	base_cap_amount = 1

	category = amenity
	
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_authority = auth_warshipgirls	
		}
		NOT = { has_modifier = slave_colony }
	}
	building_sets = {
		government
		urban
		entertainment
		resort
	}
	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			AND = {
				buildings_stop_pop = yes
				owner = { is_ai = yes }
			}
		}
	}
	
	planet_modifier = {
		job_barthroom_worker_add = 400
		planet_stability_add = 5
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
			sr_pantsu = 50
		}
		upkeep = {
			energy = 15
			sr_pantsu = 1
			exotic_gases = 1
		}
	}
	
	prerequisites = {
		"tech_building_wsg_bathhouse"
	}
	
	triggered_desc = {
		text = job_barthroom_worker_effect_desc
	}

	ai_weight = {
		factor = 10
		modifier = {
			factor = 10
			free_amenities <= 0
		}
	}
	
	ai_resource_production = {
		energy = 5
		trigger = { always = yes }
	}
}
